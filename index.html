const CODE = "1710";

let input = "";



function pressKey(k) {

if (input.length >= 4) return;

input += k;

updateDots(false);

if (input.length === 4) setTimeout(checkCode, 80);

}



function delKey() {

input = input.slice(0, -1);

updateDots(false);

}



function updateDots(err) {

for (let i = 0; i < 4; i++) {

const d = document.getElementById("d" + i);

d.className = "dot";

if (err) d.classList.add("error");

else if (i < input.length) d.classList.add("filled");

}

}



function checkCode() {

if (input === CODE) {

document.getElementById("lock").style.opacity = "0";

document.getElementById("lock").style.transition = "opacity .4s";

document.getElementById("lock").style.display = "none";

document.getElementById("app").style.display = "block";

renderTable(sites);

document.getElementById("countBadge").textContent = sites.length + " site" + (sites.length > 1 ? "s" : "");

}, 400);

} else {

updateDots(true);

setTimeout(() => { input = ""; updateDots(false); }, 700);

}

}

// ── TABLE ─────────────────────────────────────────────────────────────

function decodeB64(b64) {

try { return atob(b64); } catch { return "#"; }

}



function codeClass(code) {

const c = parseInt(code[0]) || 0;

return "code code-" + c;

}



function renderTable(data) {

const tbody = document.getElementById("tbody");

if (!data.length) {

tbody.innerHTML = '<tr><td colspan="4" class="no-results">Aucun résultat</td></tr>';

return;

}

tbody.innerHTML = data.map(s => {

const url = decodeB64(s.b64);

const display = url.replace(/^https?:\/\//, "").substring(0, 42) + (url.length > 45 ? "…" : "");

return `<tr>

<td class="num">${String(s.num).padStart(2,"0")}</td>

<td class="name">${s.nom}</td>

<td><span class="${codeClass(s.code)}">${s.code}</span></td>

<td class="link-cell"><a href="${url}" target="_blank">${display}</a></td>

</tr>`;

}).join("");

}



function filterSites() {

const q = document.getElementById("search").value.trim().toLowerCase();

if (!q) { renderTable(sites); return; }

if (/^\d{3}$/.test(q)) {

renderTable(sites.filter(s => s.code === q));

} else {

renderTable(sites.filter(s => s.nom.toLowerCase().includes(q)));

}

}



// ── CONVERTISSEUR ────────────────────────────────────────────────────

function encode() {

const raw = document.getElementById("rawInput").value;

document.getElementById("b64Input").value = btoa(unescape(encodeURIComponent(raw)));

}

function decode() {

try {

const b64 = document.getElementById("b64Input").value.trim();

document.getElementById("rawInput").value = decodeURIComponent(escape(atob(b64)));

} catch { document.getElementById("rawInput").value = "⚠ Base64 invalide"; }

}

function clearConv() {

document.getElementById("rawInput").value = "";

document.getElementById("b64Input").value = "";

}
